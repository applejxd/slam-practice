50 void featureTracking(const Mat &img_1, const Mat &img_2, vector<Point2f> &points1, vector<Point2f> &points2,
51  vector<uchar> &status) {
52  //this function automatically gets rid of points for which tracking fails
53  vector<float> err;
54  Size winSize = Size(21, 21);
55  TermCriteria criteria = TermCriteria(TermCriteria::COUNT + TermCriteria::EPS, 30, 0.01);
56  calcOpticalFlowPyrLK(img_1, img_2, points1, points2,
57  status, err, winSize, 3, criteria, 0, 0.001);
58 
59  //getting rid of points for which the KLT tracking failed or those who have gone outside the frame
60  int indexCorrection = 0;
61  for (int i = 0; i < status.size(); i++) {
62  Point2f pt = points2.at(i - indexCorrection);
63  // invalid optical flow or outside the frame
64  if ((status.at(i) == 0) || (pt.x < 0) || (pt.y < 0)) {
65  if ((pt.x < 0) || (pt.y < 0)) {
66  // define outside points also as invalid
67  status.at(i) = 0;
68  }
69  // remove from feature points list
70  points1.erase(points1.begin() + (i - indexCorrection));
71  points2.erase(points2.begin() + (i - indexCorrection));
72  // shift index because of remove points
73  indexCorrection++;
74  }
75 
76  }

78 }
